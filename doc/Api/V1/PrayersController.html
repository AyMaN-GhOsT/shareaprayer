def index
  last = params[:last].blank? ? Time.now.utc + 1.second : Time.parse(params[:last])
  @prayers = Prayer.feed(last)

  respond_to do |format|
    format.json  { render :json => @prayers }
    format.xml  { render :xml => @prayers }
  end
end
